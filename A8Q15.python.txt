import tkinter as tk
from tkinter import messagebox

def check_number():
    try:
        n = int(entry.get())
        choice = var.get()

        if choice == "Prime":
            if n > 1 and all(n % i != 0 for i in range(2, int(n**0.5)+1)):
                messagebox.showinfo("Result", f"{n} is Prime")
            else:
                messagebox.showinfo("Result", f"{n} is not Prime")

        elif choice == "Perfect":
            sum_div = sum(i for i in range(1, n) if n % i == 0)
            if sum_div == n:
                messagebox.showinfo("Result", f"{n} is Perfect")
            else:
                messagebox.showinfo("Result", f"{n} is not Perfect")

        elif choice == "Armstrong":
            digits = list(map(int, str(n)))
            if sum(d**len(digits) for d in digits) == n:
                messagebox.showinfo("Result", f"{n} is Armstrong")
            else:
                messagebox.showinfo("Result", f"{n} is not Armstrong")

    except ValueError:
        messagebox.showerror("Error", "Please enter a valid integer!")

win = tk.Tk()
win.title("Number Checker")
win.geometry("400x250")

tk.Label(win, text="Enter a number:").pack(pady=5)
entry = tk.Entry(win)
entry.pack(pady=5)

var = tk.StringVar(value="Prime")
tk.Radiobutton(win, text="Prime", variable=var, value="Prime").pack()
tk.Radiobutton(win, text="Perfect", variable=var, value="Perfect").pack()
tk.Radiobutton(win, text="Armstrong", variable=var, value="Armstrong").pack()

tk.Button(win, text="Check", command=check_number).pack(pady=10)

win.mainloop()
